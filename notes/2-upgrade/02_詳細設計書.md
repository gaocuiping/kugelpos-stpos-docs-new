```python
def generate_manifest_for_device(
    tenant_id: str,
    store_code: str,
    device_type: str,
    device_id: str,
    target_version: str,
    services: List[str]
) -> Dict:
    """
    特定の店舗・デバイス向けのManifestを生成
    """
    manifest = {
        "manifest_version": "1.0",
        "device_type": device_type,
        "device_id": device_id,
        "tenant_id": tenant_id,
        "store_code": store_code,
        "target_version": target_version,
        "created_at": datetime.now().isoformat(),
        "artifacts": [],
        "container_registry": get_container_registry_config(),
        "container_images": []
    }

    # 1. スクリプト情報を取得
    script_name = f"{device_type}-startup.sh"
    script_info = get_script_info_from_blob(script_name, target_version)
    manifest["artifacts"].append(script_info)

    # 2. 各サービスのイメージ情報を取得
    for service in services:
        image_info = get_image_info_from_acr(service, target_version)
        manifest["container_images"].append(image_info)

    # 3. Blob Storageに保存
    save_manifest_to_blob(manifest, tenant_id, store_code, device_type, device_id)

    return manifest
```

#### 1.1.2 スクリプト情報取得

```python
def get_script_info_from_blob(script_name: str, version: str) -> Dict:
    """Azure Blob Storageからスクリプト情報を取得"""
    blob_client = BlobServiceClient.from_connection_string(BLOB_CONN_STR)
    blob_path = f"scripts/{script_name}/{version}/{script_name}"

    # Blob情報を取得
    blob = blob_client.get_blob_client(
        container="kugelpos-artifacts",
        blob=blob_path
    )
    properties = blob.get_blob_properties()

    # チェックサムを計算（または metadata から取得）
    checksum = properties.metadata.get('sha256')
    if not checksum:
        # Blobをダウンロードしてチェックサムを計算
        data = blob.download_blob().readall()
        checksum = f"sha256:{hashlib.sha256(data).hexdigest()}"

    return {
        "type": "script",
        "name": script_name,
        "version": version,
        "url": "https://sync.kugelpos.cloud/api/v1/artifact-management/download",
        "checksum": checksum,
        "size": properties.size,
        "destination": f"/opt/kugelpos/scripts/{script_name}",
        "permissions": "0755",
        "priority": 1,
        "required": True
    }
```

```python
def get_image_info_from_acr(service: str, version: str) -> Dict:
    """ACRからイメージ情報を取得"""
    acr_client = ContainerRegistryClient(
        endpoint=ACR_ENDPOINT,
        credential=ACR_CREDENTIAL
    )

    repository = f"production/services/{service}"

    # イメージのダイジェストを取得
    manifest_props = acr_client.get_manifest_properties(repository, version)
    digest = manifest_props.digest

    return {
        "service": service,
        "version": version,
        "registry": "masakugel.azurecr.io",
        "image": f"{repository}:{version}",
        "checksum": digest
    }
```

### 1.2 更新処理フロー

#### 1.2.1 起動時の処理フロー

```
起動
 │
 ├─ Phase 1: 環境確認
 │   ├─ ディスク容量確認（最低10GB）
 │   ├─ ネットワーク接続確認
 │   └─ Docker動作確認
 │
 ├─ Phase 2: スクリプト自己更新チェック
 │   ├─ Sync ServiceからManifest取得
 │   ├─ 自分自身（startup.sh）の更新確認
 │   ├─ 更新あり？
 │   │   ├─ Yes: ダウンロード→検証→バックアップ→上書き→再起動
 │   │   └─ No: 次のPhaseへ
 │   └─ （ここで再起動される場合あり）
 │
 ├─ Phase 3: 通常ファイルの更新
 │   ├─ Manifestの全artifactsを取得
 │   ├─ priority順にソート
 │   ├─ 各ファイルをSync Service APIからダウンロード
 │   │   ├─ チェックサム検証
 │   │   ├─ 配置先にコピー
 │   │   ├─ 権限設定
 │   │   └─ インストールコマンド実行（必要な場合）
 │   └─ 全ファイル処理完了
 │
 ├─ Phase 4: コンテナイメージの更新
 │   ├─ ACRにログイン
 │   ├─ Manifestのcontainer_imagesを取得
 │   ├─ 各イメージをpull
 │   │   ├─ Edge: ACRから直接pull
 │   │   └─ POS: Edge Registry → ACR（フォールバック）
 │   ├─ ダイジェスト検証
 │   └─ docker-compose.yml更新
 │
 ├─ Phase 5: サービス再起動とヘルスチェック
 │   ├─ docker-compose down/up
 │   ├─ ヘルスチェック（最大5分）
 │   ├─ 成功？
 │   │   ├─ Yes: Sync Serviceに完了通知
 │   │   └─ No: 自動ロールバック
 │   └─ post_installアクション実行
 │
 └─ 起動完了
```

#### 1.2.2 スクリプト自己更新の実装

```bash
#!/bin/bash

SCRIPT_VERSION="v1.2.2"
SCRIPT_PATH="$0"

# 1. Manifestを取得
MANIFEST=$(curl -X POST "${SYNC_API_URL}/artifact-management/check" \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "X-Device-ID: ${DEVICE_ID}" \
  -H "X-Device-Type: ${DEVICE_TYPE}" \
  -H "Content-Type: application/json" \
  -d "{
    \"device_type\": \"${DEVICE_TYPE}\",
    \"device_id\": \"${DEVICE_ID}\",
    \"current_version\": \"${SCRIPT_VERSION}\"
  }")

# 2. 自分自身のエントリを検索
SCRIPT_NAME=$(basename "$SCRIPT_PATH")
SCRIPT_ARTIFACT=$(echo "$MANIFEST" | jq ".artifacts[] | select(.name == \"${SCRIPT_NAME}\")")

if [ -z "$SCRIPT_ARTIFACT" ]; then
  echo "No update for script"
  exit 0
fi

# 3. バージョン比較
NEW_VERSION=$(echo "$SCRIPT_ARTIFACT" | jq -r '.version')
if [ "$NEW_VERSION" == "$SCRIPT_VERSION" ]; then
  echo "Script is up to date"
  exit 0
fi

echo "Updating script from ${SCRIPT_VERSION} to ${NEW_VERSION}"

# 4. ダウンロード
URL=$(echo "$SCRIPT_ARTIFACT" | jq -r '.url')
CHECKSUM=$(echo "$SCRIPT_ARTIFACT" | jq -r '.checksum')

curl -X POST "$URL" \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "X-Device-ID: ${DEVICE_ID}" \
  -H "X-Device-Type: ${DEVICE_TYPE}" \
  -H "Content-Type: application/json" \
  -d "{
    \"artifact_type\": \"script\",
    \"artifact_name\": \"${SCRIPT_NAME}\",
    \"version\": \"${NEW_VERSION}\",
    \"checksum\": \"${CHECKSUM}\"
  }" \
  -o /tmp/startup.sh.new

# 5. チェックサム検証
ACTUAL_CHECKSUM=$(sha256sum /tmp/startup.sh.new | cut -d' ' -f1)
if [ "sha256:$ACTUAL_CHECKSUM" != "$CHECKSUM" ]; then
  echo "ERROR: Checksum mismatch"
  rm -f /tmp/startup.sh.new
  exit 1
fi

# 6. バックアップ（前バージョンのみ）
BACKUP_PATH="${SCRIPT_PATH}.backup"
cp "$SCRIPT_PATH" "$BACKUP_PATH"

# 7. 上書き
mv /tmp/startup.sh.new "$SCRIPT_PATH"
chmod +x "$SCRIPT_PATH"

# 8. 再起動（exec で現プロセスを置き換え）
echo "Script updated successfully. Restarting..."
exec "$SCRIPT_PATH" "$@"
```

#### 1.2.3 通常ファイルのダウンロード処理

```bash
download_and_install_artifacts() {
  local manifest="$1"

  # artifacts を priority 順にソート
  local sorted_artifacts=$(echo "$manifest" | jq '.artifacts | sort_by(.priority)')

  # 各ファイルを処理
  echo "$sorted_artifacts" | jq -c '.[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local type=$(echo "$artifact" | jq -r '.type')
    local version=$(echo "$artifact" | jq -r '.version')
    local url=$(echo "$artifact" | jq -r '.url')
    local checksum=$(echo "$artifact" | jq -r '.checksum')
    local destination=$(echo "$artifact" | jq -r '.destination')
    local permissions=$(echo "$artifact" | jq -r '.permissions // "0644"')
    local required=$(echo "$artifact" | jq -r '.required // true')

    echo "Processing: $name ($type) version $version"

    # ダウンロード
    local temp_file="/tmp/${name}.download"
    if ! download_artifact "$type" "$name" "$version" "$checksum" "$temp_file"; then
      if [ "$required" == "true" ]; then
        echo "ERROR: Failed to download required artifact: $name"
        return 1
      else
        echo "WARNING: Failed to download optional artifact: $name"
        continue
      fi
    fi

    # チェックサム検証
    if ! verify_checksum "$temp_file" "$checksum"; then
      echo "ERROR: Checksum mismatch for $name"
      rm -f "$temp_file"
      return 1
    fi

    # バックアップ
    if [ -f "$destination" ]; then
      cp "$destination" "${destination}.backup"
    fi

    # インストール
    install_artifact "$temp_file" "$destination" "$permissions" "$type"

    # post_install コマンド実行
    local post_cmd=$(echo "$artifact" | jq -r '.post_install_command // empty')
    if [ -n "$post_cmd" ]; then
      echo "Running post-install command: $post_cmd"
      eval "$post_cmd"
    fi

    rm -f "$temp_file"
  done
}

download_artifact() {
  local type="$1"
  local name="$2"
  local version="$3"
  local checksum="$4"
  local output="$5"

  curl -X POST "${SYNC_API_URL}/artifact-management/download" \
    -H "Authorization: Bearer ${API_KEY}" \
    -H "X-Device-ID: ${DEVICE_ID}" \
    -H "X-Device-Type: ${DEVICE_TYPE}" \
    -H "Content-Type: application/json" \
    -d "{
      \"artifact_type\": \"${type}\",
      \"artifact_name\": \"${name}\",
      \"version\": \"${version}\",
      \"checksum\": \"${checksum}\"
    }" \
    -o "$output" \
    --fail \
    --silent \
    --show-error
}

verify_checksum() {
  local file="$1"
  local expected_checksum="$2"

  local actual_checksum=$(sha256sum "$file" | cut -d' ' -f1)

  if [ "sha256:$actual_checksum" == "$expected_checksum" ]; then
    return 0
  else
    return 1
  fi
}

install_artifact() {
  local source="$1"
  local destination="$2"
  local permissions="$3"
  local type="$4"

  case "$type" in
    script|image|config)
      # 直接コピー
      mkdir -p "$(dirname "$destination")"
      cp "$source" "$destination"
      chmod "$permissions" "$destination"
      ;;
    module)
      # 拡張子に応じて処理
      case "$source" in
        *.whl)
          pip install "$source"
          ;;
        *.deb)
          dpkg -i "$source"
          ;;
        *.tar.gz)
          tar -xzf "$source" -C "$(dirname "$destination")"
          ;;
      esac
      ;;
  esac
}
```

#### 1.2.4 コンテナイメージの更新処理

```bash
update_container_images() {
  local manifest="$1"

  # 1. ACRにログイン
  local acr_url=$(echo "$manifest" | jq -r '.container_registry.url')
  local acr_username=$(cat /opt/kugelpos/secrets/ACR_USERNAME)
  local acr_password=$(cat /opt/kugelpos/secrets/ACR_PASSWORD)

  docker login "$acr_url" -u "$acr_username" -p "$acr_password"

  # 2. 現在のチェックサムを読み込み
  local current_checksums_file="/opt/kugelpos/cache/current-checksums.json"
  local current_checksums="{}"
  if [ -f "$current_checksums_file" ]; then
    current_checksums=$(cat "$current_checksums_file")
  fi

  # 3. 各イメージを処理
  echo "$manifest" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local version=$(echo "$image" | jq -r '.version')
    local image_name=$(echo "$image" | jq -r '.image')
    local new_digest=$(echo "$image" | jq -r '.checksum')

    # 現在のダイジェストを取得
    local current_digest=$(echo "$current_checksums" | jq -r ".\"${service}\" // empty")

    # ダイジェストが一致していればスキップ
    if [ "$current_digest" == "$new_digest" ]; then
      echo "Skipping ${service}: already up to date"
      continue
    fi

    echo "Pulling updated image: ${service} (${version})"

    # イメージをpull
    if ! docker pull "$image_name"; then
      echo "ERROR: Failed to pull image: ${image_name}"
      return 1
    fi

    # ダイジェストを検証
    local actual_digest=$(docker inspect --format='{{index .RepoDigests 0}}' "$image_name" | cut -d'@' -f2)
    if [ "$actual_digest" != "$new_digest" ]; then
      echo "ERROR: Image digest mismatch for ${service}"
      echo "  Expected: ${new_digest}"
      echo "  Actual: ${actual_digest}"
      return 1
    fi

    echo "Image ${service} updated successfully"
  done

  # 4. docker-compose.ymlを更新
  update_docker_compose_file "$manifest"
}

update_docker_compose_file() {
  local manifest="$1"
  local compose_file="/opt/kugelpos/docker-compose.yml"

  # バックアップ
  cp "$compose_file" "${compose_file}.backup"

  # 各サービスのイメージを更新
  echo "$manifest" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local image_name=$(echo "$image" | jq -r '.image')

    # yq を使用して YAML を更新
    yq eval ".services.${service}.image = \"${image_name}\"" -i "$compose_file"
  done
}
```

#### 1.3.1 ヘルスチェック処理

```bash
health_check() {
  local timeout=300  # 5分
  local interval=5   # 5秒ごとにチェック
  local elapsed=0

  echo "Starting health check..."

  while [ $elapsed -lt $timeout ]; do
    local all_healthy=true

    # 各サービスのヘルスエンドポイントをチェック
    for service in account terminal master-data cart report journal stock; do
      local port=$(get_service_port "$service")

      if ! curl -s -f "http://localhost:${port}/health" > /dev/null; then
        echo "Service $service is not ready yet..."
        all_healthy=false
        break
      fi
    done

    if [ "$all_healthy" == "true" ]; then
      echo "All services are healthy"
      return 0
    fi

    sleep $interval
    elapsed=$((elapsed + interval))
  done

  echo "ERROR: Health check timeout after ${timeout} seconds"
  return 1
}

get_service_port() {
  case "$1" in
    account) echo 8000 ;;
    terminal) echo 8001 ;;
    master-data) echo 8002 ;;
    cart) echo 8003 ;;
    report) echo 8004 ;;
    journal) echo 8005 ;;
    stock) echo 8006 ;;
  esac
}
```

#### 1.3.2 ロールバック処理

```bash
rollback() {
  local reason="$1"

  echo "Starting rollback due to: $reason"

  # 1. docker-compose.ymlをロールバック
  local compose_file="/opt/kugelpos/docker-compose.yml"
  if [ -f "${compose_file}.backup" ]; then
    mv "${compose_file}.backup" "$compose_file"
  fi

  # 2. スクリプトをロールバック
  if [ -f "${SCRIPT_PATH}.backup" ]; then
    mv "${SCRIPT_PATH}.backup" "$SCRIPT_PATH"
  fi

  # 3. その他のファイルをロールバック
  find /opt/kugelpos -name "*.backup" | while read -r backup_file; do
    local original="${backup_file%.backup}"
    mv "$backup_file" "$original"
  done

  # 4. サービスを再起動
  docker-compose down
  docker-compose up -d

  # 5. ヘルスチェック
  if health_check; then
    echo "Rollback successful"

    # 6. エラーをSync Serviceに通知
    notify_rollback "$reason"

    return 0
  else
    echo "ERROR: Rollback failed"
    return 1
  fi
}

notify_rollback() {
  local reason="$1"

  curl -X POST "${SYNC_API_URL}/artifact-management/complete" \
    -H "Authorization: Bearer ${API_KEY}" \
    -H "X-Device-ID: ${DEVICE_ID}" \
    -H "X-Device-Type: ${DEVICE_TYPE}" \
    -H "Content-Type: application/json" \
    -d "{
      \"device_type\": \"${DEVICE_TYPE}\",
      \"device_id\": \"${DEVICE_ID}\",
      \"update_results\": [{
        \"status\": \"rollback\",
        \"reason\": \"${reason}\"
      }],
      \"timestamp\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"
    }"
}
```

## 3. 実装設計

### 3.1 Sync Service API実装

#### 3.1.1 ファイルダウンロードAPI

```python
from fastapi import APIRouter, HTTPException, Header, Depends
from fastapi.responses import StreamingResponse
from azure.storage.blob import BlobServiceClient
import logging

router = APIRouter()
logger = logging.getLogger(__name__)

@router.post("/artifact-management/download")
async def download_artifact(
    request: DownloadRequest,
    device_id: str = Header(..., alias="X-Device-ID"),
    device_type: str = Header(..., alias="X-Device-Type"),
    authorization: str = Header(..., alias="Authorization"),
    blob_service: BlobServiceClient = Depends(get_blob_service)
):
    # 1. 認証・認可チェック
    api_key = authorization.replace("Bearer ", "")
    if not await verify_device_access(device_id, device_type, api_key):
        logger.warning(f"Unauthorized download attempt: {device_id}")
        raise HTTPException(status_code=401, detail="Unauthorized")

    # 2. アクセスログ記録
    await log_download_request(
        device_id=device_id,
        artifact_name=request.artifact_name,
        version=request.version
    )

    # 3. Blob Storageからファイル取得
    blob_path = f"{request.artifact_type}s/{request.artifact_name}/{request.version}/{request.artifact_name}"
    blob_client = blob_service.get_blob_client(
        container="kugelpos-artifacts",
        blob=blob_path
    )

    # 4. ファイル存在チェック
    if not await blob_client.exists():
        logger.error(f"Artifact not found: {blob_path}")
        raise HTTPException(status_code=404, detail="Artifact not found")

    # 5. Blob情報を取得
    blob_properties = await blob_client.get_blob_properties()

    # 6. ストリーミングでダウンロード
    async def stream_blob():
        stream = await blob_client.download_blob()
        async for chunk in stream.chunks():
            yield chunk

    logger.info(f"Streaming file: {request.artifact_name} to {device_id}")

    return StreamingResponse(
        stream_blob(),
        media_type="application/octet-stream",
        headers={
            "Content-Disposition": f"attachment; filename={request.artifact_name}",
            "X-Checksum": request.checksum,
            "X-File-Size": str(blob_properties.size)
        }
    )


class DownloadRequest(BaseModel):
    artifact_type: str
    artifact_name: str
    version: str
    checksum: str


async def verify_device_access(device_id: str, device_type: str, api_key: str) -> bool:
```

```python
@router.post("/artifact-management/check")
async def check_updates(
    request: CheckRequest,
    device_id: str = Header(..., alias="X-Device-ID"),
    device_type: str = Header(..., alias="X-Device-Type"),
    authorization: str = Header(None, alias="Authorization")

    # 1. 認証（POS端末のみ）
    if device_type == "pos":
        if not authorization:

        api_key = authorization.replace("Bearer ", "")
        if not await verify_pos_terminal(request.device_id, api_key):
            raise HTTPException(status_code=401, detail="Invalid credentials")

    # 2. Manifestを取得
    manifest = await get_manifest_for_device(
        request.device_id,
        request.device_type
    )

    if not manifest:
        raise HTTPException(status_code=404, detail="No manifest found for this device")

    # 3. 差分を計算
    diff_manifest = calculate_diff(manifest, request.current_checksums)

    return diff_manifest


def calculate_diff(manifest: Dict, current_checksums: Dict) -> Dict:
    """現在のチェックサムと比較して差分のみを返却"""
    diff_manifest = manifest.copy()
    diff_manifest["updates_required"] = False
    diff_manifest["artifacts"] = []
    diff_manifest["container_images"] = []
    diff_manifest["no_change"] = []

    # Artifactsの差分チェック
    for artifact in manifest.get("artifacts", []):
        name = artifact["name"]
        new_checksum = artifact["checksum"]
        current_checksum = current_checksums.get(name)

        if current_checksum != new_checksum:
            # 更新が必要
            artifact["action"] = "update"
            artifact["reason"] = "checksum_mismatch"
            artifact["current_checksum"] = current_checksum
            artifact["new_checksum"] = new_checksum
            diff_manifest["artifacts"].append(artifact)
            diff_manifest["updates_required"] = True
        else:
            # 変更なし
            diff_manifest["no_change"].append({
                "name": name,
                "checksum": new_checksum,
                "message": "Already up to date"
            })

    # Container Imagesの差分チェック
    for image in manifest.get("container_images", []):
        service = image["service"]
        new_checksum = image["checksum"]
        current_checksum = current_checksums.get(service)

        if current_checksum != new_checksum:
            # 更新が必要
            image["action"] = "update"
            image["reason"] = "version_upgrade"
            image["current_checksum"] = current_checksum
            image["new_checksum"] = new_checksum
            diff_manifest["container_images"].append(image)
            diff_manifest["updates_required"] = True
        else:
            # 変更なし
            diff_manifest["no_change"].append({
                "service": service,
                "checksum": new_checksum,
                "message": "Already up to date"
            })

    return diff_manifest

    device_id: str
    current_version: str = None
```

```bash
# リトライロジック（指数バックオフ）
retry_with_backoff() {
  local max_attempts=3
  local timeout=30
  local attempt=1
  local backoff=1

  while [ $attempt -le $max_attempts ]; do
    echo "Attempt $attempt of $max_attempts..."

    if timeout $timeout "$@"; then
      return 0
    fi

    if [ $attempt -lt $max_attempts ]; then
      echo "Failed. Retrying in ${backoff} seconds..."
      sleep $backoff
      backoff=$((backoff * 2))
    fi

    attempt=$((attempt + 1))
  done

  echo "ERROR: All attempts failed"
  return 1
}

# 使用例
if ! retry_with_backoff curl -f "${SYNC_API_URL}/artifact-management/check"; then
  echo "Failed to connect to Sync Service. Using cached manifest..."
  use_cached_manifest
fi
```

```bash
# チェックサム検証とリトライ
download_with_checksum_verification() {
  local url="$1"
  local expected_checksum="$2"
  local output="$3"
  local max_retries=3

  for i in $(seq 1 $max_retries); do
    echo "Download attempt $i..."

    if curl -f -o "$output" "$url"; then
      local actual_checksum=$(sha256sum "$output" | cut -d' ' -f1)

      if [ "sha256:$actual_checksum" == "$expected_checksum" ]; then
        echo "Checksum verified successfully"
        return 0
      else
        echo "ERROR: Checksum mismatch (attempt $i)"
        echo "  Expected: $expected_checksum"
        echo "  Actual: sha256:$actual_checksum"
        rm -f "$output"
      fi
    else
      echo "ERROR: Download failed (attempt $i)"
    fi
  done

  echo "ERROR: All download attempts failed"
  return 1
}
```
