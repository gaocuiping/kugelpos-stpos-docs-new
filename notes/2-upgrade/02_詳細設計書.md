```python
def generate_manifest_for_device(
    tenant_id: str,
    store_code: str,
    device_type: str,
    device_id: str,
    target_version: str,
    services: List[str]
) -> Dict:
    """
    特定の店舗・デバイス向けのManifestを生成
    """
    manifest = {
        "manifest_version": "1.0",
        "device_type": device_type,
        "device_id": device_id,
        "tenant_id": tenant_id,
        "store_code": store_code,
        "target_version": target_version,
        "created_at": datetime.now().isoformat(),
        "artifacts": [],
        "container_registry": get_container_registry_config(),
        "container_images": []
    }

    # 1. スクリプト情報を取得（データソースルーティング付き）
    script_name = f"{device_type}-startup.sh"
    script_info = get_artifact_info_with_routing(
        artifact_type="script",
        artifact_name=script_name,
        version=target_version,
        device_type=device_type,
        store_code=store_code
    )
    manifest["artifacts"].append(script_info)

    # 2. 各サービスのイメージ情報を取得（データソースルーティング付き）
    for service in services:
        image_info = get_image_info_with_routing(
            service,
            target_version,
            device_type,
            store_code
        )
        manifest["container_images"].append(image_info)

    # 3. Blob Storageに保存
    save_manifest_to_blob(manifest, tenant_id, store_code, device_type, device_id)

    return manifest
```

#### 1.1.2 アーティファクト情報取得（データソースルーティング付き）

```python
def get_artifact_info_with_routing(
    artifact_type: str,
    artifact_name: str,
    version: str,
    device_type: str,
    store_code: str
) -> Dict:
    """
    データソースルーティング情報を含むアーティファクト情報を取得

    重要: 全てのファイル取得はAPI経由で行われます

    - Edge端末: data_source = "cloud" (Cloud Sync Service API経由で取得)
    - POS端末: data_source = "edge" (Edge Sync Service API経由で取得、Cloud Sync APIへフォールバック)

    Edge端末にはSync Serviceが稼働しており、ローカルにキャッシュされたファイルをAPI経由で配信します
    """
    # Azure Blob Storageから基本情報を取得
    blob_client = BlobServiceClient.from_connection_string(BLOB_CONN_STR)

    # アーティファクトタイプに応じてパスを決定
    if artifact_type == "script":
        blob_path = f"scripts/{artifact_name}/{version}/{artifact_name}"
        destination = f"/opt/kugelpos/scripts/{artifact_name}"
        permissions = "0755"
    elif artifact_type == "config":
        blob_path = f"configs/{artifact_name}/{version}/{artifact_name}"
        destination = f"/opt/kugelpos/config/{artifact_name}"
        permissions = "0644"
    elif artifact_type == "module":
        blob_path = f"modules/{artifact_name}"
        destination = None  # インストールコマンドで決定
        permissions = "0644"
    else:
        blob_path = f"{artifact_type}s/{artifact_name}/{version}/{artifact_name}"
        destination = f"/opt/kugelpos/{artifact_type}/{artifact_name}"
        permissions = "0644"

    # Blob情報を取得
    blob = blob_client.get_blob_client(
        container="kugelpos-artifacts",
        blob=blob_path
    )
    properties = blob.get_blob_properties()

    # チェックサムを取得
    checksum = properties.metadata.get('sha256')
    if not checksum:
        data = blob.download_blob().readall()
        checksum = f"sha256:{hashlib.sha256(data).hexdigest()}"

    # デバイスタイプに応じてデータソースを決定
    if device_type == "edge":
        # Edge端末は常にクラウド（Sync Service経由）から取得
        return {
            "type": artifact_type,
            "name": artifact_name,
            "version": version,
            "data_source": "cloud",
            "primary_url": f"https://sync.kugelpos.cloud/api/v1/artifact-management/download",
            "fallback_url": None,
            "checksum": checksum,
            "size": properties.size,
            "destination": destination,
            "permissions": permissions,
            "priority": 1 if artifact_type == "script" else 2,
            "required": True
        }
    elif device_type == "pos":
        # POS端末はEdge端末のSync Service APIから取得、失敗時はCloud Sync Serviceへフォールバック
        edge_id = get_edge_id_for_store(store_code)
        edge_sync_service_url = f"http://{edge_id}:8007"

        return {
            "type": artifact_type,
            "name": artifact_name,
            "version": version,
            "data_source": "edge",
            "primary_url": f"{edge_sync_service_url}/api/v1/artifact-management/download",
            "fallback_url": f"https://sync.kugelpos.cloud/api/v1/artifact-management/download",
            "checksum": checksum,
            "size": properties.size,
            "destination": destination,
            "permissions": permissions,
            "priority": 1 if artifact_type == "script" else 2,
            "required": True
        }

def get_edge_id_for_store(store_code: str) -> str:
    """
    店舗コードから対応するEdge端末のIDを取得

    例: store-tokyo-001 → edge-tokyo-001
    """
    # データベースまたは設定から店舗のEdge端末情報を取得
    # 実装例:
    edge_id = f"edge-{store_code.replace('store-', '')}"
    return edge_id
```

#### 1.1.3 イメージ情報取得（データソースルーティング付き）

```python
def get_image_info_with_routing(
    service: str,
    version: str,
    device_type: str,
    store_code: str
) -> Dict:
    """
    データソースルーティング情報を含むイメージ情報を取得

    - Edge端末: data_source = "cloud" (ACRから直接取得)
    - POS端末: data_source = "edge" (Edge Registryから取得、ACRへフォールバック)
    """
    acr_client = ContainerRegistryClient(
        endpoint=ACR_ENDPOINT,
        credential=ACR_CREDENTIAL
    )

    repository = f"production/services/{service}"

    # ACRからイメージのダイジェストを取得
    manifest_props = acr_client.get_manifest_properties(repository, version)
    digest = manifest_props.digest

    # デバイスタイプに応じてデータソースを決定
    if device_type == "edge":
        # Edge端末は常にクラウド（ACR）から直接取得
        return {
            "service": service,
            "version": version,
            "data_source": "cloud",
            "primary_registry": "masakugel.azurecr.io",
            "primary_image": f"{repository}:{version}",
            "checksum": digest
        }
    elif device_type == "pos":
        # POS端末はEdge Registryから取得、失敗時はACRへフォールバック
        edge_registry = get_edge_registry_for_store(store_code)

        return {
            "service": service,
            "version": version,
            "data_source": "edge",
            "primary_registry": f"{edge_registry}:5000",
            "primary_image": f"{service}:{version}",
            "fallback_registry": "masakugel.azurecr.io",
            "fallback_image": f"{repository}:{version}",
            "checksum": digest
        }


def get_edge_registry_for_store(store_code: str) -> str:
    """
    店舗コードから対応するEdge Registryのホスト名を取得

    例: store-tokyo-001 → edge-tokyo-001
    """
    # データベースまたは設定から店舗のEdge Registry情報を取得
    # 実装例:
    edge_id = f"edge-{store_code.replace('store-', '')}"
    return edge_id
```

### 1.2 更新処理フロー

#### 1.2.1 起動時の処理フロー

```
起動
 │
 ├─ Phase 1: 環境確認
 │   ├─ ディスク容量確認（最低10GB）
 │   ├─ ネットワーク接続確認
 │   └─ Docker動作確認
 │
 ├─ Phase 2: スクリプト自己更新チェック
 │   ├─ Sync ServiceからManifest取得
 │   ├─ 自分自身（startup.sh）の更新確認
 │   ├─ 更新あり？
 │   │   ├─ Yes: ダウンロード→検証→バックアップ→上書き→再起動
 │   │   └─ No: 次のPhaseへ
 │   └─ （ここで再起動される場合あり）
 │
 ├─ Phase 3: 通常ファイルの更新
 │   ├─ Manifestの全artifactsを取得
 │   ├─ priority順にソート
 │   ├─ 各ファイルをSync Service APIからダウンロード
 │   │   ├─ チェックサム検証
 │   │   ├─ 配置先にコピー
 │   │   ├─ 権限設定
 │   │   └─ インストールコマンド実行（必要な場合）
 │   └─ 全ファイル処理完了
 │
 ├─ Phase 4: コンテナイメージの更新
 │   ├─ ACRにログイン
 │   ├─ Manifestのcontainer_imagesを取得
 │   ├─ 各イメージをpull
 │   │   ├─ Edge: ACRから直接pull
 │   │   └─ POS: Edge Registry → ACR（フォールバック）
 │   ├─ ダイジェスト検証
 │   └─ docker-compose.yml更新
 │
 ├─ Phase 5: サービス再起動とヘルスチェック
 │   ├─ docker-compose down/up
 │   ├─ ヘルスチェック（最大5分）
 │   ├─ 成功？
 │   │   ├─ Yes: Sync Serviceに完了通知
 │   │   └─ No: 自動ロールバック（直前のバージョンへ）
 │   └─ post_installアクション実行
 │
 └─ 起動完了
```

#### 1.2.2 スクリプト自己更新の実装

```bash
#!/bin/bash

SCRIPT_VERSION="v1.2.2"
SCRIPT_PATH="$0"

# 1. Manifestを取得
MANIFEST=$(curl -X POST "${SYNC_API_URL}/artifact-management/check" \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "X-Device-ID: ${DEVICE_ID}" \
  -H "X-Device-Type: ${DEVICE_TYPE}" \
  -H "Content-Type: application/json" \
  -d "{
    \"device_type\": \"${DEVICE_TYPE}\",
    \"device_id\": \"${DEVICE_ID}\",
    \"current_version\": \"${SCRIPT_VERSION}\"
  }")

# 2. 自分自身のエントリを検索
SCRIPT_NAME=$(basename "$SCRIPT_PATH")
SCRIPT_ARTIFACT=$(echo "$MANIFEST" | jq ".artifacts[] | select(.name == \"${SCRIPT_NAME}\")")

if [ -z "$SCRIPT_ARTIFACT" ]; then
  echo "No update for script"
  exit 0
fi

# 3. バージョン比較
NEW_VERSION=$(echo "$SCRIPT_ARTIFACT" | jq -r '.version')
if [ "$NEW_VERSION" == "$SCRIPT_VERSION" ]; then
  echo "Script is up to date"
  exit 0
fi

echo "Updating script from ${SCRIPT_VERSION} to ${NEW_VERSION}"

# 4. ダウンロード
URL=$(echo "$SCRIPT_ARTIFACT" | jq -r '.url')
CHECKSUM=$(echo "$SCRIPT_ARTIFACT" | jq -r '.checksum')

curl -X POST "$URL" \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "X-Device-ID: ${DEVICE_ID}" \
  -H "X-Device-Type: ${DEVICE_TYPE}" \
  -H "Content-Type: application/json" \
  -d "{
    \"artifact_type\": \"script\",
    \"artifact_name\": \"${SCRIPT_NAME}\",
    \"version\": \"${NEW_VERSION}\",
    \"checksum\": \"${CHECKSUM}\"
  }" \
  -o /tmp/startup.sh.new

# 5. チェックサム検証
ACTUAL_CHECKSUM=$(sha256sum /tmp/startup.sh.new | cut -d' ' -f1)
if [ "sha256:$ACTUAL_CHECKSUM" != "$CHECKSUM" ]; then
  echo "ERROR: Checksum mismatch"
  rm -f /tmp/startup.sh.new
  exit 1
fi

# 6. バックアップ（前バージョンのみ）
BACKUP_PATH="${SCRIPT_PATH}.backup"
cp "$SCRIPT_PATH" "$BACKUP_PATH"

# 7. 上書き
mv /tmp/startup.sh.new "$SCRIPT_PATH"
chmod +x "$SCRIPT_PATH"

# 8. 再起動（exec で現プロセスを置き換え）
echo "Script updated successfully. Restarting..."
exec "$SCRIPT_PATH" "$@"
```

#### 1.2.3 通常ファイルのダウンロード処理

```bash
download_and_install_artifacts() {
  local manifest="$1"

  # artifacts を priority 順にソート
  local sorted_artifacts=$(echo "$manifest" | jq '.artifacts | sort_by(.priority)')

  # 各ファイルを処理
  echo "$sorted_artifacts" | jq -c '.[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local type=$(echo "$artifact" | jq -r '.type')
    local version=$(echo "$artifact" | jq -r '.version')
    local url=$(echo "$artifact" | jq -r '.url')
    local checksum=$(echo "$artifact" | jq -r '.checksum')
    local destination=$(echo "$artifact" | jq -r '.destination')
    local permissions=$(echo "$artifact" | jq -r '.permissions // "0644"')
    local required=$(echo "$artifact" | jq -r '.required // true')

    echo "Processing: $name ($type) version $version"

    # ダウンロード
    local temp_file="/tmp/${name}.download"
    if ! download_artifact "$type" "$name" "$version" "$checksum" "$temp_file"; then
      if [ "$required" == "true" ]; then
        echo "ERROR: Failed to download required artifact: $name"
        return 1
      else
        echo "WARNING: Failed to download optional artifact: $name"
        continue
      fi
    fi

    # チェックサム検証
    if ! verify_checksum "$temp_file" "$checksum"; then
      echo "ERROR: Checksum mismatch for $name"
      rm -f "$temp_file"
      return 1
    fi

    # バックアップ
    if [ -f "$destination" ]; then
      cp "$destination" "${destination}.backup"
    fi

    # インストール
    install_artifact "$temp_file" "$destination" "$permissions" "$type"

    # post_install コマンド実行
    local post_cmd=$(echo "$artifact" | jq -r '.post_install_command // empty')
    if [ -n "$post_cmd" ]; then
      echo "Running post-install command: $post_cmd"
      eval "$post_cmd"
    fi

    rm -f "$temp_file"
  done
}

download_artifact() {
  local type="$1"
  local name="$2"
  local version="$3"
  local checksum="$4"
  local output="$5"

  curl -X POST "${SYNC_API_URL}/artifact-management/download" \
    -H "Authorization: Bearer ${API_KEY}" \
    -H "X-Device-ID: ${DEVICE_ID}" \
    -H "X-Device-Type: ${DEVICE_TYPE}" \
    -H "Content-Type: application/json" \
    -d "{
      \"artifact_type\": \"${type}\",
      \"artifact_name\": \"${name}\",
      \"version\": \"${version}\",
      \"checksum\": \"${checksum}\"
    }" \
    -o "$output" \
    --fail \
    --silent \
    --show-error
}

verify_checksum() {
  local file="$1"
  local expected_checksum="$2"

  local actual_checksum=$(sha256sum "$file" | cut -d' ' -f1)

  if [ "sha256:$actual_checksum" == "$expected_checksum" ]; then
    return 0
  else
    return 1
  fi
}

install_artifact() {
  local source="$1"
  local destination="$2"
  local permissions="$3"
  local type="$4"

  case "$type" in
    script|image|config)
      # 直接コピー
      mkdir -p "$(dirname "$destination")"
      cp "$source" "$destination"
      chmod "$permissions" "$destination"
      ;;
    module)
      # 拡張子に応じて処理
      case "$source" in
        *.whl)
          pip install "$source"
          ;;
        *.deb)
          dpkg -i "$source"
          ;;
        *.tar.gz)
          tar -xzf "$source" -C "$(dirname "$destination")"
          ;;
      esac
      ;;
  esac
}
```

#### 1.2.4 コンテナイメージの更新処理

```bash
update_container_images() {
  local manifest="$1"

  # 1. レジストリ認証情報を準備
  local acr_username=$(cat /opt/kugelpos/secrets/ACR_USERNAME)
  local acr_password=$(cat /opt/kugelpos/secrets/ACR_PASSWORD)

  # 2. 現在のチェックサムを読み込み
  local current_checksums_file="/opt/kugelpos/cache/current-checksums.json"
  local current_checksums="{}"
  if [ -f "$current_checksums_file" ]; then
    current_checksums=$(cat "$current_checksums_file")
  fi

  # 3. 各イメージを処理
  echo "$manifest" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local version=$(echo "$image" | jq -r '.version')
    local data_source=$(echo "$image" | jq -r '.data_source')
    local new_digest=$(echo "$image" | jq -r '.checksum')

    # 現在のダイジェストを取得
    local current_digest=$(echo "$current_checksums" | jq -r ".\"${service}\" // empty")

    # ダイジェストが一致していればスキップ
    if [ "$current_digest" == "$new_digest" ]; then
      echo "Skipping ${service}: already up to date"
      continue
    fi

    echo "Pulling updated image: ${service} (${version}) from ${data_source}"

    # データソースに応じてイメージを取得
    local image_pulled=false
    local final_image_name=""

    if [ "$data_source" == "edge" ]; then
      # Edge Registryから取得を試行、失敗時はACRへフォールバック
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')
      local fallback_registry=$(echo "$image" | jq -r '.fallback_registry')
      local fallback_image=$(echo "$image" | jq -r '.fallback_image')

      local primary_full_image="${primary_registry}/${primary_image}"
      local fallback_full_image="${fallback_registry}/${fallback_image}"

      echo "Attempting to pull from Edge Registry: ${primary_full_image}"

      # Edge Registryから取得を試行（認証なし、プライベートネットワーク想定）
      if docker pull "$primary_full_image" 2>/dev/null; then
        echo "Successfully pulled from Edge Registry"
        final_image_name="$primary_full_image"
        image_pulled=true
      else
        echo "Failed to pull from Edge Registry, falling back to ACR: ${fallback_full_image}"

        # ACRにログインしてフォールバック
        docker login "$fallback_registry" -u "$acr_username" -p "$acr_password"

        if docker pull "$fallback_full_image"; then
          echo "Successfully pulled from ACR (fallback)"
          final_image_name="$fallback_full_image"
          image_pulled=true
        else
          echo "ERROR: Failed to pull image from both Edge Registry and ACR"
          return 1
        fi
      fi

    elif [ "$data_source" == "cloud" ]; then
      # ACRから直接取得
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')
      local primary_full_image="${primary_registry}/${primary_image}"

      echo "Pulling from ACR: ${primary_full_image}"

      # ACRにログイン
      docker login "$primary_registry" -u "$acr_username" -p "$acr_password"

      if docker pull "$primary_full_image"; then
        echo "Successfully pulled from ACR"
        final_image_name="$primary_full_image"
        image_pulled=true
      else
        echo "ERROR: Failed to pull image from ACR"
        return 1
      fi
    fi

    # イメージが正常に取得できたか確認
    if [ "$image_pulled" != "true" ]; then
      echo "ERROR: Failed to pull image: ${service}"
      return 1
    fi

    # ダイジェストを検証
    local actual_digest=$(docker inspect --format='{{index .RepoDigests 0}}' "$final_image_name" | cut -d'@' -f2)
    if [ "$actual_digest" != "$new_digest" ]; then
      echo "ERROR: Image digest mismatch for ${service}"
      echo "  Expected: ${new_digest}"
      echo "  Actual: ${actual_digest}"
      return 1
    fi

    # ローカルタグ付け（docker-composeで使用するため）
    local local_tag="kugelpos/${service}:${version}"
    docker tag "$final_image_name" "$local_tag"

    echo "Image ${service} updated successfully and tagged as ${local_tag}"
  done

  # 4. docker-compose.ymlを更新
  update_docker_compose_file "$manifest"
}

update_docker_compose_file() {
  local manifest="$1"
  local compose_file="/opt/kugelpos/docker-compose.yml"

  # バックアップ
  cp "$compose_file" "${compose_file}.backup"

  # 各サービスのイメージを更新（ローカルタグを使用）
  echo "$manifest" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local version=$(echo "$image" | jq -r '.version')
    local local_tag="kugelpos/${service}:${version}"

    # yq を使用して YAML を更新
    yq eval ".services.${service}.image = \"${local_tag}\"" -i "$compose_file"
  done
}
```

#### 1.2.5 フェーズ別詳細実装

機能要件書で定義された9つの更新フェーズと実装の対応関係を以下に示します。

**フェーズ1: バージョンチェック**

```bash
check_version() {
  echo "[Phase 1] Version check started"

  # 現在のバージョン情報を取得
  local current_version=$(cat /opt/kugelpos/VERSION || echo "v0.0.0")

  # JWT Token取得（認証済みの場合はスキップ）
  if [ -z "$JWT_TOKEN" ]; then
    authenticate_device
  fi

  # Manifestを取得
  local response=$(curl -s -X POST "${SYNC_API_URL}/artifact-management/check" \
    -H "Authorization: Bearer ${JWT_TOKEN}" \
    -H "Content-Type: application/json" \
    -d "{
      \"device_type\": \"${DEVICE_TYPE}\",
      \"device_id\": \"${DEVICE_ID}\",
      \"current_version\": \"${current_version}\"
    }")

  # 更新の有無を判定
  local target_version=$(echo "$response" | jq -r '.target_version')

  if [ "$target_version" == "$current_version" ]; then
    echo "[Phase 1] No update required. Current version: ${current_version}"
    return 1  # 更新不要
  fi

  echo "[Phase 1] Update available: ${current_version} → ${target_version}"

  # Manifestをグローバル変数に保存
  MANIFEST="$response"
  TARGET_VERSION="$target_version"

  return 0  # 更新が必要
}
```

**実装のポイント:**
- このフェーズでは**ファイルのダウンロードは行わない**
- 更新の必要性を判定するだけ
- 失敗時は現在のバージョンで起動を継続

```bash
    local data_source=$(echo "$image" | jq -r '.data_source')

    echo "  Downloading: ${service}"

    if [ "$data_source" == "edge" ]; then
      # Edge Registryから試行
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')
      local primary_full="${primary_registry}/${primary_image}"

      if docker pull "$primary_full" 2>/dev/null; then
        echo "    Downloaded from Edge Registry: ${primary_full}"
        echo "image:${service}:${primary_full}" >> "$download_log"
      else
        # フォールバック: ACRから取得
        local fallback_registry=$(echo "$image" | jq -r '.fallback_registry')
        local fallback_image=$(echo "$image" | jq -r '.fallback_image')
        local fallback_full="${fallback_registry}/${fallback_image}"

        docker login "$fallback_registry" -u "$ACR_USERNAME" -p "$ACR_PASSWORD"
        if docker pull "$fallback_full"; then
          echo "    Downloaded from ACR (fallback): ${fallback_full}"
          echo "image:${service}:${fallback_full}" >> "$download_log"
        else
          echo "ERROR: Failed to download ${service}"
          cleanup_temp_dir "$temp_dir"
          return 1
        fi
      fi
    elif [ "$data_source" == "cloud" ]; then
      # ACRから直接取得
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')
      local primary_full="${primary_registry}/${primary_image}"

      docker login "$primary_registry" -u "$ACR_USERNAME" -p "$ACR_PASSWORD"
      if docker pull "$primary_full"; then
        echo "    Downloaded from ACR: ${primary_full}"
        echo "image:${service}:${primary_full}" >> "$download_log"
      else
        echo "ERROR: Failed to download ${service}"
        cleanup_temp_dir "$temp_dir"
        return 1
      fi
    fi
  done

  # 2. アーティファクトファイルをダウンロード（data_sourceルーティング対応）
  echo "[Phase 2] Downloading artifacts..."
  echo "$MANIFEST" | jq -c '.artifacts[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local type=$(echo "$artifact" | jq -r '.type')
    local data_source=$(echo "$artifact" | jq -r '.data_source')

    echo "  Downloading: ${name} (${type}) from ${data_source}"

    local temp_file="${temp_dir}/${name}"
    local downloaded=false

    if [ "$data_source" == "edge" ]; then
      # Edge Sync Service APIから取得を試行
      local primary_url=$(echo "$artifact" | jq -r '.primary_url')
      local fallback_url=$(echo "$artifact" | jq -r '.fallback_url')

      echo "    Trying primary (Edge Sync API): ${primary_url}"

      # Edge Sync Service APIへリクエスト送信
      if curl -s -X POST "$primary_url" \
        -H "Authorization: Bearer ${JWT_TOKEN}" \
        -H "Content-Type: application/json" \
        -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
        -o "$temp_file" 2>/dev/null; then
        echo "    Downloaded from Edge Sync Service API"
        downloaded=true
      else
        echo "    Failed to download from Edge Sync API, falling back to Cloud Sync API: ${fallback_url}"

        # フォールバック: Cloud Sync Service APIへリクエスト
        if curl -s -X POST "$fallback_url" \
          -H "Authorization: Bearer ${JWT_TOKEN}" \
          -H "Content-Type: application/json" \
          -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
          -o "$temp_file"; then
          echo "    Downloaded from Cloud Sync Service API (fallback)"
          downloaded=true
        else
          echo "ERROR: Failed to download ${name} from both Edge and Cloud Sync APIs"
          cleanup_temp_dir "$temp_dir"
          return 1
        fi
      fi

    elif [ "$data_source" == "cloud" ]; then
      # Cloud Sync Service API経由で直接取得
      local primary_url=$(echo "$artifact" | jq -r '.primary_url')

      echo "    Downloading from Cloud Sync API: ${primary_url}"

      if curl -s -X POST "$primary_url" \
        -H "Authorization: Bearer ${JWT_TOKEN}" \
        -H "Content-Type: application/json" \
        -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
        -o "$temp_file"; then
        echo "    Downloaded from Cloud Sync Service API"
        downloaded=true
      else
        echo "ERROR: Failed to download ${name} from Cloud Sync API"
        cleanup_temp_dir "$temp_dir"
        return 1
      fi
    fi

    if [ "$downloaded" == "true" ]; then
      echo "${type}:${name}:${temp_file}" >> "$download_log"
    else
      echo "ERROR: Failed to download ${name}"
      cleanup_temp_dir "$temp_dir"
      return 1
    fi
  done

  echo "[Phase 2] All downloads completed"

  # ダウンロードディレクトリをグローバル変数に保存
  UPDATE_TEMP_DIR="$temp_dir"

  return 0
}
```

**実装のポイント:**
- すべてのファイルを一時ディレクトリにダウンロード
- **この段階ではまだ適用しない**（サービスは稼働継続）
- ダウンロードしたファイルのリストを記録
- 失敗時は一時ディレクトリを削除してクリーンアップ
- **重要**: コンテナイメージ以外のファイルは全てSync Service API経由で取得（ファイルシステムやストレージへの直接アクセスは行わない）
- Edge端末向け: Cloud Sync Service APIへリクエスト
- POS端末向け: Edge Sync Service APIへリクエスト → 失敗時はCloud Sync Service APIへフォールバック

**フェーズ3: 検証**

```bash
verify_downloads() {
  echo "[Phase 3] Verification started"

  # 1. コンテナイメージのダイジェスト検証
  echo "[Phase 3] Verifying container images..."
  echo "$MANIFEST" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local expected_checksum=$(echo "$image" | jq -r '.checksum')

    # ダウンロードログから実際のイメージ名を取得
    local actual_image=$(grep "^image:${service}:" "${UPDATE_TEMP_DIR}/downloaded.log" | cut -d':' -f3-)

    # 実際のダイジェストを取得
    local actual_digest=$(docker inspect --format='{{index .RepoDigests 0}}' "$actual_image" | cut -d'@' -f2)

    if [ "$actual_digest" != "$expected_checksum" ]; then
      echo "ERROR: Checksum mismatch for ${service}"
      echo "  Expected: ${expected_checksum}"
      echo "  Actual: ${actual_digest}"
      return 1
    fi

    echo "  ✓ ${service}: checksum verified"
  done

  # 2. ファイルのチェックサム検証
  echo "[Phase 3] Verifying files..."
  echo "$MANIFEST" | jq -c '.artifacts[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local expected_checksum=$(echo "$artifact" | jq -r '.checksum')

    local temp_file="${UPDATE_TEMP_DIR}/${name}"

    if [ ! -f "$temp_file" ]; then
      echo "ERROR: File not found: ${temp_file}"
      return 1
    fi

    local actual_checksum=$(sha256sum "$temp_file" | awk '{print $1}')

    # "sha256:" プレフィックスを追加
    actual_checksum="sha256:${actual_checksum}"

    if [ "$actual_checksum" != "$expected_checksum" ]; then
      echo "ERROR: Checksum mismatch for ${name}"
      echo "  Expected: ${expected_checksum}"
      echo "  Actual: ${actual_checksum}"
      return 1
    fi

    echo "  ✓ ${name}: checksum verified"
  done

  echo "[Phase 3] All verifications passed"
  return 0
}
```

**実装のポイント:**
- すべてのダウンロード済みファイルのチェックサムを検証
- **検証が完了するまで適用は行わない**
- 1つでも検証失敗したら即座に中止

**フェーズ5: 適用準備**

```bash
prepare_application() {
  echo "[Phase 5] Preparation started"

  # 1. docker-compose.ymlを更新
  echo "[Phase 5] Updating docker-compose.yml..."
  local compose_file="/opt/kugelpos/docker-compose.yml"

  echo "$MANIFEST" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local version=$(echo "$image" | jq -r '.version')
    local local_tag="kugelpos/${service}:${version}"

    # ローカルタグ付け
    local actual_image=$(grep "^image:${service}:" "${UPDATE_TEMP_DIR}/downloaded.log" | cut -d':' -f3-)
    docker tag "$actual_image" "$local_tag"

    # docker-compose.ymlを更新
    yq eval ".services.${service}.image = \"${local_tag}\"" -i "$compose_file"

    echo "  ✓ Updated: ${service} → ${local_tag}"
  done

  # 2. スクリプトファイルを更新（まだ適用しない）
  echo "[Phase 5] Preparing script files..."
  echo "$MANIFEST" | jq -c '.artifacts[] | select(.type == "script")' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local temp_file="${UPDATE_TEMP_DIR}/${name}"

    # 実行権限を付与（まだ本番ディレクトリにはコピーしない）
    chmod +x "$temp_file"

    echo "  ✓ Prepared: ${name}"
  done

  # 3. その他の設定ファイルを準備
  echo "[Phase 5] Preparing configuration files..."
  echo "$MANIFEST" | jq -c '.artifacts[] | select(.type == "config")' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local destination=$(echo "$artifact" | jq -r '.destination')

    # 宛先ディレクトリを作成
    mkdir -p "$(dirname "$destination")"

    echo "  ✓ Prepared: ${name}"
  done

  echo "[Phase 5] Preparation completed"
  echo "[Phase 5] NOTE: Services are still running with old version"

  return 0
}
```

**実装のポイント:**
- docker-compose.ymlは更新するが、**サービスはまだ再起動しない**
- スクリプトファイルは準備するが、**本番ディレクトリには配置しない**
- **この段階までサービスは稼働継続**

```bash
  return 0
}
```

```bash
    sleep 5
    elapsed=$((elapsed + 5))

  echo "ERROR: [Phase 7] Failed to start all containers"
  return 1
}
```

**フェーズ8: ヘルスチェック**

```bash
health_check() {
  echo "[Phase 8] Health check started"

  # 起動直後は待機
  echo "[Phase 8] Waiting 30 seconds for services to initialize..."
  sleep 30

  local max_attempts=3
  local attempt=0

  while [ $attempt -lt $max_attempts ]; do
    attempt=$((attempt + 1))
    echo "[Phase 8] Health check attempt ${attempt}/${max_attempts}"

    local all_healthy=true

    # 各サービスのヘルスチェック
    for service in account terminal master-data cart report journal stock; do
      local port=$(get_service_port "$service")

      if curl -s -f "http://localhost:${port}/health" > /dev/null 2>&1; then
        echo "  ✓ ${service}: healthy"
      else
        echo "  ✗ ${service}: unhealthy"
        all_healthy=false
      fi
    done

    if [ "$all_healthy" == "true" ]; then
      echo "[Phase 8] All services are healthy"
      return 0
    fi

    if [ $attempt -lt $max_attempts ]; then
      echo "[Phase 8] Retrying in 10 seconds..."
      sleep 10
    fi
  done

  echo "ERROR: [Phase 8] Health check failed after ${max_attempts} attempts"
  return 1
}
```

**実装のポイント:**
- 30秒待機してからヘルスチェック開始
- 最大3回、10秒間隔でリトライ
- **失敗したらフェーズ10（自動ロールバック）へ**
- **ヘルスチェックはダウンタイムの終了地点**

**フェーズ10: 自動ロールバック（エラー時のみ）**

```bash
auto_rollback() {
  echo "[Phase 10] AUTO ROLLBACK started"
  echo "[Phase 10] Rolling back to previous version: ${CURRENT_VERSION}"

  # 1. サービスを停止
  echo "[Phase 10] Stopping failed services..."
  docker-compose -f /opt/kugelpos/docker-compose.yml down --timeout 10

  # 2. docker-compose.ymlを復元
  if [ -f "${BACKUP_DIR}/docker-compose.yml.backup" ]; then
    cp "${BACKUP_DIR}/docker-compose.yml.backup" /opt/kugelpos/docker-compose.yml
    echo "  ✓ Restored: docker-compose.yml"
  fi

  # 3. スクリプトを復元
  local script_name="${DEVICE_TYPE}-startup.sh"
  if [ -f "${BACKUP_DIR}/${script_name}.backup" ]; then
    cp "${BACKUP_DIR}/${script_name}.backup" "/opt/kugelpos/${script_name}"
    chmod +x "/opt/kugelpos/${script_name}"
    echo "  ✓ Restored: ${script_name}"
  fi

  # 4. 設定ファイルを復元
  if [ -d "${BACKUP_DIR}/config.backup" ]; then
    rm -rf /opt/kugelpos/config
    cp -r "${BACKUP_DIR}/config.backup" /opt/kugelpos/config
    echo "  ✓ Restored: config/"
  fi

  # 5. 直前のバージョンでサービスを起動
  echo "[Phase 10] Starting services with previous version: ${CURRENT_VERSION}..."
  docker-compose -f /opt/kugelpos/docker-compose.yml up -d

  # 6. ヘルスチェック
  sleep 30
  if health_check; then
    echo "[Phase 10] Auto rollback completed successfully"

    # 自動ロールバック完了を通知
    notify_auto_rollback

    return 0
  else
    echo "CRITICAL: [Phase 10] Auto rollback failed! Manual intervention required!"
    send_emergency_alert
    return 1
  fi
}

notify_auto_rollback() {
  curl -s -X POST "${SYNC_API_URL}/artifact-management/apply-complete" \
    -H "Authorization: Bearer ${JWT_TOKEN}" \
    -H "Content-Type: application/json" \
    -d "{
      \"device_type\": \"${DEVICE_TYPE}\",
      \"device_id\": \"${DEVICE_ID}\",
      \"from_version\": \"${CURRENT_VERSION}\",
      \"to_version\": \"${TARGET_VERSION}\",
      \"apply_status\": \"auto_rollback\",
      \"rolled_back_to_version\": \"${CURRENT_VERSION}\",
      \"error_message\": \"Update failed and automatically rolled back to previous version\",
      \"updated_at\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"
    }"
}
```

**実装のポイント:**
- フェーズ7またはフェーズ8で失敗した場合のみ実行
- 常に「直前のバージョン」に自動ロールバック（任意バージョンへの手動ロールバックは不可）
- バックアップから完全に復元
- 直前のバージョンでサービスを再起動
- Sync Serviceに`auto_rollback`ステータスで通知
- **自動ロールバックも失敗した場合は緊急アラート**

**運用方針:**
- ファイル内容に不備があった場合は、修正版を新しいバージョンとして配信・適用する
- 管理者による手動ロールバック指示機能は提供しない（運用をシンプルに保つ）

**メインフロー（全フェーズの統合）**

```bash
main_update_flow() {
  # フェーズ1: バージョンチェック
  if ! check_version; then
    echo "No update required"
    return 0
  fi

  # フェーズ2: ダウンロード
  if ! download_all_updates; then
    echo "ERROR: Download failed"
    return 1
  fi

  # フェーズ3: 検証
  if ! verify_downloads; then
    echo "ERROR: Verification failed"
    cleanup_temp_dir "$UPDATE_TEMP_DIR"
    return 1
  fi

  # フェーズ4: バックアップ
  if ! backup_current_state; then
    echo "ERROR: Backup failed"
    cleanup_temp_dir "$UPDATE_TEMP_DIR"
    return 1
  fi

  # フェーズ5: 適用準備
  if ! prepare_application; then
    echo "ERROR: Preparation failed"
    cleanup_temp_dir "$UPDATE_TEMP_DIR"
    return 1
  fi

  # ここまでサービスは稼働継続
  echo "===== SERVICES WILL BE STOPPED ====="

  # フェーズ6: サービス停止（ダウンタイム開始）
  if ! stop_services; then
    echo "WARNING: Service stop had issues"
  fi

  # フェーズ7: サービス起動
  if ! start_services; then
    echo "ERROR: Service startup failed, initiating auto rollback"
    auto_rollback
    return 1
  fi

  # フェーズ8: ヘルスチェック
  if ! health_check; then
    echo "ERROR: Health check failed, initiating auto rollback"
    auto_rollback
    return 1
  fi

  # ダウンタイム終了
  echo "===== SERVICES ARE HEALTHY ====="

  # フェーズ9: 完了通知
  notify_completion

  echo "Update completed successfully!"
  return 0
}
```

**重要な時系列:**
1. **フェーズ1-5**: サービスは稼働継続（ダウンロード、検証、準備のみ）
2. **フェーズ6**: サービス停止 → **ダウンタイム開始**
3. **フェーズ7-8**: サービス起動とヘルスチェック → **ダウンタイム中**
4. **フェーズ8成功**: **ダウンタイム終了**
5. **フェーズ9**: 完了処理

**通常のダウンタイム:** 1-3分（フェーズ6開始 → フェーズ8成功）

#### 1.2.6 2段階更新モードの実装

2段階更新機能により、ダウンロードと適用を別々のステップで実行できます。

**コマンドライン引数:**

```bash
#!/bin/bash
# pos-startup.sh または edge-startup.sh

# 引数解析
UPDATE_MODE="normal"  # デフォルトは通常モード

while [[ $# -gt 0 ]]; do
  case $1 in
    --download-only)
      UPDATE_MODE="download_only"
      shift
      ;;
    --apply-only)
      UPDATE_MODE="apply_only"
      shift
      ;;
    *)
      echo "Unknown option: $1"
      exit 1
      ;;
  esac
```

```bash
download_only_mode() {
  echo "[Download-Only Mode] Starting..."

  # Phase 1: バージョンチェック
  check_version || {
    echo "No updates available"
    return 0
  }

  TARGET_VERSION=$(echo "$MANIFEST" | jq -r '.target_version')
  PENDING_DIR="/opt/kugelpos/pending-updates/${TARGET_VERSION}"

  # 既にダウンロード済みかチェック
  if [ -f "${PENDING_DIR}/status.json" ]; then
    local status=$(cat "${PENDING_DIR}/status.json" | jq -r '.download_status')
    if [ "$status" == "completed" ]; then
      echo "Version ${TARGET_VERSION} already downloaded. Skipping."
      return 0
    fi
  fi

  # ディレクトリ作成
  mkdir -p "${PENDING_DIR}/artifacts"
  mkdir -p "${PENDING_DIR}/images"

  # ステータスファイル初期化
  cat > "${PENDING_DIR}/status.json" <<EOF
{
  "version": "${TARGET_VERSION}",
  "download_status": "in_progress",
  "download_started_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "download_completed_at": null,
  "verification_status": "pending",
  "ready_to_apply": false,
  "artifacts_count": 0,
  "total_size_bytes": 0
}
EOF

  # Phase 2: ダウンロード
  echo "[Phase 2] Downloading all files..."

  # 2.1 コンテナイメージをダウンロード
  download_container_images_to_cache "${PENDING_DIR}" || {
    echo "ERROR: Failed to download container images"
    cleanup_pending_dir "${PENDING_DIR}"
    return 1
  }

  # 2.2 アーティファクトをダウンロード
  download_artifacts_to_cache "${PENDING_DIR}" || {
    echo "ERROR: Failed to download artifacts"
    cleanup_pending_dir "${PENDING_DIR}"
    return 1
  }

  # Phase 3: 検証
  echo "[Phase 3] Verifying downloaded files..."
  verify_downloaded_files "${PENDING_DIR}" || {
    echo "ERROR: Verification failed"
    cleanup_pending_dir "${PENDING_DIR}"
    return 1
  }

  # Manifestを保存
  echo "$MANIFEST" > "${PENDING_DIR}/manifest.json"

  # ステータスファイル更新
  local artifacts_count=$(echo "$MANIFEST" | jq '.artifacts | length')
  local images_count=$(echo "$MANIFEST" | jq '.container_images | length')
  local total_count=$((artifacts_count + images_count))

  cat > "${PENDING_DIR}/status.json" <<EOF
{
  "version": "${TARGET_VERSION}",
  "download_status": "completed",
  "download_started_at": "$(cat ${PENDING_DIR}/status.json | jq -r '.download_started_at')",
  "download_completed_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "verification_status": "passed",
  "ready_to_apply": true,
  "artifacts_count": ${total_count},
  "total_size_bytes": $(du -sb "${PENDING_DIR}" | cut -f1)
}
EOF

  echo "[Download-Only Mode] Completed successfully"
  echo "Downloaded version: ${TARGET_VERSION}"
  echo "Ready to apply. Run with --apply-only to apply changes."

  # クラウドに完了通知
  notify_download_complete "${TARGET_VERSION}" "${PENDING_DIR}"

  return 0
}

download_container_images_to_cache() {
  local pending_dir="$1"
  local image_list="${pending_dir}/images/image-list.txt"
  > "$image_list"  # ファイル初期化

  echo "$MANIFEST" | jq -c '.container_images[]' | while read -r image; do
    local service=$(echo "$image" | jq -r '.service')
    local version=$(echo "$image" | jq -r '.version')
    local data_source=$(echo "$image" | jq -r '.data_source')

    echo "  Pulling image: ${service}:${version}"

    # イメージをpull（1.2.4のロジックと同じ）
    if [ "$data_source" == "edge" ]; then
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')
      local fallback_registry=$(echo "$image" | jq -r '.fallback_registry')
      local fallback_image=$(echo "$image" | jq -r '.fallback_image')

      if docker pull "${primary_registry}/${primary_image}" 2>/dev/null; then
        echo "${primary_registry}/${primary_image}" >> "$image_list"
      elif docker pull "${fallback_registry}/${fallback_image}"; then
        echo "${fallback_registry}/${fallback_image}" >> "$image_list"
      else
        echo "ERROR: Failed to pull ${service}"
        return 1
      fi
    elif [ "$data_source" == "cloud" ]; then
      local primary_registry=$(echo "$image" | jq -r '.primary_registry')
      local primary_image=$(echo "$image" | jq -r '.primary_image')

      if docker pull "${primary_registry}/${primary_image}"; then
        echo "${primary_registry}/${primary_image}" >> "$image_list"
      else
        echo "ERROR: Failed to pull ${service}"
        return 1
      fi
    fi
  done

  return 0
}

download_artifacts_to_cache() {
  local pending_dir="$1"

  echo "$MANIFEST" | jq -c '.artifacts[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local type=$(echo "$artifact" | jq -r '.type')
    local data_source=$(echo "$artifact" | jq -r '.data_source')
    local checksum=$(echo "$artifact" | jq -r '.checksum')

    echo "  Downloading: ${name}"

    local dest_file="${pending_dir}/artifacts/${name}"

    # API経由でダウンロード（1.2.5のロジックと同じ）
    if [ "$data_source" == "edge" ]; then
      local primary_url=$(echo "$artifact" | jq -r '.primary_url')
      local fallback_url=$(echo "$artifact" | jq -r '.fallback_url')

      if ! curl -s -X POST "$primary_url" \
        -H "Authorization: Bearer ${JWT_TOKEN}" \
        -H "Content-Type: application/json" \
        -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
        -o "$dest_file"; then

        curl -s -X POST "$fallback_url" \
          -H "Authorization: Bearer ${JWT_TOKEN}" \
          -H "Content-Type: application/json" \
          -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
          -o "$dest_file" || return 1
      fi
    elif [ "$data_source" == "cloud" ]; then
      local primary_url=$(echo "$artifact" | jq -r '.primary_url')

      curl -s -X POST "$primary_url" \
        -H "Authorization: Bearer ${JWT_TOKEN}" \
        -H "Content-Type: application/json" \
        -d "{\"artifact_type\": \"${type}\", \"artifact_name\": \"${name}\"}" \
        -o "$dest_file" || return 1
    fi

    # チェックサム検証
    local actual_checksum=$(sha256sum "$dest_file" | cut -d' ' -f1)
    local expected_checksum=$(echo "$checksum" | sed 's/^sha256://')

    if [ "$actual_checksum" != "$expected_checksum" ]; then
      echo "ERROR: Checksum mismatch for ${name}"
      return 1
    fi

    echo "    OK: ${name}"
  done

  return 0
}

verify_downloaded_files() {
  local pending_dir="$1"

  echo "Verifying all downloaded files..."

  # 全アーティファクトのチェックサムを再検証
  echo "$MANIFEST" | jq -c '.artifacts[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local checksum=$(echo "$artifact" | jq -r '.checksum')
    local file_path="${pending_dir}/artifacts/${name}"

    if [ ! -f "$file_path" ]; then
      echo "ERROR: Missing file: ${name}"
      return 1
    fi

    local actual_checksum=$(sha256sum "$file_path" | cut -d' ' -f1)
    local expected_checksum=$(echo "$checksum" | sed 's/^sha256://')

    if [ "$actual_checksum" != "$expected_checksum" ]; then
      echo "ERROR: Checksum mismatch for ${name}"
      return 1
    fi
  done

  echo "All files verified successfully"
  return 0
}
```

```bash
apply_only_mode() {
  echo "[Apply-Only Mode] Starting..."

  # ダウンロード済みファイルを検索
  local pending_base="/opt/kugelpos/pending-updates"

  if [ ! -d "$pending_base" ]; then
    echo "ERROR: No pending updates directory found"
    return 1
  fi

  # 最新バージョンを取得（複数ある場合）
  local latest_version=$(ls -1 "$pending_base" | sort -V | tail -1)

  if [ -z "$latest_version" ]; then
    echo "ERROR: No pending updates found"
    return 1
  fi

  local pending_dir="${pending_base}/${latest_version}"
  local status_file="${pending_dir}/status.json"
  local manifest_file="${pending_dir}/manifest.json"

  # ステータスチェック
  if [ ! -f "$status_file" ]; then
    echo "ERROR: Status file not found: ${status_file}"
    return 1
  fi

  local download_status=$(cat "$status_file" | jq -r '.download_status')
  local ready_to_apply=$(cat "$status_file" | jq -r '.ready_to_apply')

  if [ "$download_status" != "completed" ] || [ "$ready_to_apply" != "true" ]; then
    echo "ERROR: Pending update is not ready to apply"
    echo "  Download status: ${download_status}"
    echo "  Ready to apply: ${ready_to_apply}"
    return 1
  fi

  # Manifestを読み込み
  if [ ! -f "$manifest_file" ]; then
    echo "ERROR: Manifest file not found: ${manifest_file}"
    return 1
  fi

  MANIFEST=$(cat "$manifest_file")
  TARGET_VERSION="$latest_version"

  echo "Applying version: ${TARGET_VERSION}"

  # Phase 3: 再検証（ダウンロード後の改ざん防止）
  echo "[Phase 3] Re-verifying downloaded files..."
  verify_downloaded_files "${pending_dir}" || {
    echo "ERROR: Re-verification failed. Files may have been tampered with."
    return 1
  }

  # Phase 4: バックアップ
  echo "[Phase 4] Creating backup..."
  backup_current_version || {
    echo "ERROR: Backup failed"
    return 1
  }

  # Phase 5: 適用準備
  echo "[Phase 5] Preparing for update..."
  prepare_update_from_cache "${pending_dir}" || {
    echo "ERROR: Preparation failed"
    auto_rollback
    return 1
  }

  # Phase 6: サービス停止
  echo "[Phase 6] Stopping services..."
  echo "WARNING: Downtime starts now"
  docker-compose down || {
    echo "WARNING: Graceful shutdown failed, forcing stop"
    docker-compose down --timeout 5
  }

  # Phase 7: サービス起動
  echo "[Phase 7] Starting services with new version..."
  docker-compose up -d || {
    echo "ERROR: Failed to start services"
    auto_rollback
    return 1
  }

  # Phase 8: ヘルスチェック
  echo "[Phase 8] Running health check..."
  sleep 30  # サービス起動を待つ

  health_check || {
    echo "ERROR: Health check failed"
    auto_rollback
    return 1
  }

  echo "SUCCESS: Downtime ended. Services are healthy."

  # Phase 9: 完了通知
  echo "[Phase 9] Notifying completion..."
  notify_update_complete "${TARGET_VERSION}" "apply_only"

  # ダウンロード済みファイルをクリーンアップ
  cleanup_pending_dir "${pending_dir}"

  echo "[Apply-Only Mode] Completed successfully"
  return 0
}

prepare_update_from_cache() {
  local pending_dir="$1"

  # アーティファクトを配置
  echo "$MANIFEST" | jq -c '.artifacts[]' | while read -r artifact; do
    local name=$(echo "$artifact" | jq -r '.name')
    local destination=$(echo "$artifact" | jq -r '.destination')
    local permissions=$(echo "$artifact" | jq -r '.permissions')
    local source_file="${pending_dir}/artifacts/${name}"

    if [ ! -f "$source_file" ]; then
      echo "ERROR: Source file not found: ${source_file}"
      return 1
    fi

    echo "  Copying ${name} to ${destination}"
    cp "$source_file" "$destination"
    chmod "$permissions" "$destination"
  done

  # docker-compose.ymlを更新
  update_docker_compose_from_manifest

  return 0
}

cleanup_pending_dir() {
  local pending_dir="$1"

  if [ -d "$pending_dir" ]; then
    echo "Cleaning up: ${pending_dir}"
    rm -rf "$pending_dir"
  fi
}
```

```bash
health_check() {
  local timeout=300  # 5分
  local interval=5   # 5秒ごとにチェック
  local elapsed=0

  echo "Starting health check..."

  while [ $elapsed -lt $timeout ]; do
    local all_healthy=true

    # 各サービスのヘルスエンドポイントをチェック
    for service in account terminal master-data cart report journal stock; do
      local port=$(get_service_port "$service")

      if ! curl -s -f "http://localhost:${port}/health" > /dev/null; then
        echo "Service $service is not ready yet..."
        all_healthy=false
        break
      fi
    done

    if [ "$all_healthy" == "true" ]; then
      echo "All services are healthy"
      return 0
    fi

    sleep $interval
    elapsed=$((elapsed + interval))
  done

  echo "ERROR: Health check timeout after ${timeout} seconds"
  return 1
}

get_service_port() {
  case "$1" in
    account) echo 8000 ;;
    terminal) echo 8001 ;;
    master-data) echo 8002 ;;
    cart) echo 8003 ;;
    report) echo 8004 ;;
    journal) echo 8005 ;;
    stock) echo 8006 ;;
  esac
}
```

#### 1.3.2 自動ロールバック処理

```bash
auto_rollback() {
  local reason="$1"

  echo "Starting auto rollback to previous version due to: $reason"

  # 1. docker-compose.ymlをロールバック（直前のバージョンへ）
  local compose_file="/opt/kugelpos/docker-compose.yml"
  if [ -f "${compose_file}.backup" ]; then
    mv "${compose_file}.backup" "$compose_file"
  fi

  # 2. スクリプトをロールバック（直前のバージョンへ）
  if [ -f "${SCRIPT_PATH}.backup" ]; then
    mv "${SCRIPT_PATH}.backup" "$SCRIPT_PATH"
  fi

  # 3. その他のファイルをロールバック（直前のバージョンへ）
  find /opt/kugelpos -name "*.backup" | while read -r backup_file; do
    local original="${backup_file%.backup}"
    mv "$backup_file" "$original"
  done

  # 4. サービスを再起動（直前のバージョンで）
  docker-compose down
  docker-compose up -d

  # 5. ヘルスチェック
  if health_check; then
    echo "Auto rollback to previous version successful"

    # 6. 自動ロールバックをSync Serviceに通知
    notify_auto_rollback "$reason"

    return 0
  else
    echo "ERROR: Auto rollback failed"
    return 1
  fi
}

notify_auto_rollback() {
  local reason="$1"

  curl -X POST "${SYNC_API_URL}/artifact-management/apply-complete" \
    -H "Authorization: Bearer ${API_KEY}" \
    -H "X-Device-ID: ${DEVICE_ID}" \
    -H "X-Device-Type: ${DEVICE_TYPE}" \
    -H "Content-Type: application/json" \
    -d "{
      \"device_type\": \"${DEVICE_TYPE}\",
      \"device_id\": \"${DEVICE_ID}\",
      \"from_version\": \"${CURRENT_VERSION}\",
      \"to_version\": \"${TARGET_VERSION}\",
      \"apply_status\": \"auto_rollback\",
      \"rolled_back_to_version\": \"${CURRENT_VERSION}\",
      \"error_message\": \"${reason}\",
      \"timestamp\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"
    }"
}
```

**運用方針:**
- 常に「直前のバージョン」に自動ロールバック
- 任意バージョンへの手動ロールバック機能は提供しない
- ファイル内容に不備があった場合は、修正版を新しいバージョンとして配信・適用する

### 3.1 Sync Service API実装

#### 3.1.1 ファイルダウンロードAPI

```python
from fastapi import APIRouter, HTTPException, Header, Depends
from fastapi.responses import StreamingResponse
from azure.storage.blob import BlobServiceClient
import logging

router = APIRouter()
logger = logging.getLogger(__name__)

@router.post("/artifact-management/download")
async def download_artifact(
    request: DownloadRequest,
    device_id: str = Header(..., alias="X-Device-ID"),
    device_type: str = Header(..., alias="X-Device-Type"),
    authorization: str = Header(..., alias="Authorization"),
    blob_service: BlobServiceClient = Depends(get_blob_service)
):
    # 1. 認証・認可チェック（JWT Token検証）
    token = authorization.replace("Bearer ", "")
    if not await verify_jwt_token(token, device_id, device_type):
        logger.warning(f"Unauthorized download attempt: {device_id}")
        raise HTTPException(status_code=401, detail="Unauthorized")

    # 2. アクセスログ記録
    await log_download_request(
        device_id=device_id,
        artifact_name=request.artifact_name,
        version=request.version
    )

    # 3. Blob Storageからファイル取得
    blob_path = f"{request.artifact_type}s/{request.artifact_name}/{request.version}/{request.artifact_name}"
    blob_client = blob_service.get_blob_client(
        container="kugelpos-artifacts",
        blob=blob_path
    )

    # 4. ファイル存在チェック
    if not await blob_client.exists():
        logger.error(f"Artifact not found: {blob_path}")
        raise HTTPException(status_code=404, detail="Artifact not found")

    # 5. Blob情報を取得
    blob_properties = await blob_client.get_blob_properties()

    # 6. ストリーミングでダウンロード
    async def stream_blob():
        stream = await blob_client.download_blob()
        async for chunk in stream.chunks():
            yield chunk

        stream_blob(),
        media_type="application/octet-stream",
        headers={
            "Content-Disposition": f"attachment; filename={request.artifact_name}",
            "X-Checksum": request.checksum,
            "X-File-Size": str(blob_properties.size)
        }
    )

async def verify_jwt_token(token: str, device_id: str, device_type: str) -> bool:
    """JWTトークンを検証（Edge/POS共通）"""
    try:
        # JWTトークンをデコード
        payload = jwt.decode(
            token,
            JWT_SECRET_KEY,
            algorithms=["HS256"]
        )

        # トークンに含まれるdevice_idとリクエストのdevice_idが一致するか確認
        if payload.get("edge_id") != device_id:
            logger.warning(f"Device ID mismatch: token={payload.get('edge_id')}, request={device_id}")
            return False

        # device_typeが一致するか確認
        if payload.get("device_type") != device_type:
            logger.warning(f"Device type mismatch: token={payload.get('device_type')}, request={device_type}")
            return False

        # トークンの有効期限を確認
        if payload.get("exp") < datetime.now().timestamp():
            logger.warning(f"Token expired for device: {device_id}")
            return False

        return True

        logger.warning(f"Expired token for device: {device_id}")
        return False
    except jwt.InvalidTokenError:
        logger.warning(f"Invalid token for device: {device_id}")
        return False
```

#### 3.1.2 差分更新API

```python
@router.post("/artifact-management/check")
async def check_updates(
    request: CheckRequest,
    device_id: str = Header(..., alias="X-Device-ID"),
    device_type: str = Header(..., alias="X-Device-Type"),
    authorization: str = Header(..., alias="Authorization")
):
    # 1. 認証（JWT Token検証 - Edge/POS共通）
    if not authorization:
        raise HTTPException(status_code=401, detail="Authorization required")

    token = authorization.replace("Bearer ", "")
    if not await verify_jwt_token(token, device_id, device_type):
        raise HTTPException(status_code=401, detail="Invalid credentials")

    # 2. Manifestを取得
    manifest = await get_manifest_for_device(
        request.device_id,
        request.device_type
    )

    if not manifest:
        raise HTTPException(status_code=404, detail="No manifest found for this device")

    # 3. 差分を計算
    diff_manifest = calculate_diff(manifest, request.current_checksums)

    return diff_manifest


def calculate_diff(manifest: Dict, current_checksums: Dict) -> Dict:
    """現在のチェックサムと比較して差分のみを返却"""
    diff_manifest = manifest.copy()
    diff_manifest["updates_required"] = False
    diff_manifest["artifacts"] = []
    diff_manifest["container_images"] = []
    diff_manifest["no_change"] = []

    # Artifactsの差分チェック
    for artifact in manifest.get("artifacts", []):
        name = artifact["name"]
        new_checksum = artifact["checksum"]
        current_checksum = current_checksums.get(name)

        if current_checksum != new_checksum:
            # 更新が必要
            artifact["action"] = "update"
            artifact["reason"] = "checksum_mismatch"
            artifact["current_checksum"] = current_checksum
            artifact["new_checksum"] = new_checksum
            diff_manifest["artifacts"].append(artifact)
            diff_manifest["updates_required"] = True
        else:
            # 変更なし
            diff_manifest["no_change"].append({
                "name": name,
                "checksum": new_checksum,
                "message": "Already up to date"
            })

    # Container Imagesの差分チェック
    for image in manifest.get("container_images", []):
        service = image["service"]
        new_checksum = image["checksum"]
        current_checksum = current_checksums.get(service)

        if current_checksum != new_checksum:
            # 更新が必要
            image["action"] = "update"
            image["reason"] = "version_upgrade"
            image["current_checksum"] = current_checksum
            image["new_checksum"] = new_checksum
            diff_manifest["container_images"].append(image)
            diff_manifest["updates_required"] = True
        else:
            # 変更なし
            diff_manifest["no_change"].append({
                "service": service,
                "checksum": new_checksum,
                "message": "Already up to date"
            })

    return diff_manifest
```

```bash
# リトライロジック（指数バックオフ）
retry_with_backoff() {
  local max_attempts=3
  local timeout=30
  local attempt=1
  local backoff=1

  while [ $attempt -le $max_attempts ]; do
    echo "Attempt $attempt of $max_attempts..."

    if timeout $timeout "$@"; then
      return 0
    fi

    if [ $attempt -lt $max_attempts ]; then
      echo "Failed. Retrying in ${backoff} seconds..."
      sleep $backoff
      backoff=$((backoff * 2))
    fi

    attempt=$((attempt + 1))
  done

  echo "ERROR: All attempts failed"
  return 1
}

# 使用例
if ! retry_with_backoff curl -f "${SYNC_API_URL}/artifact-management/check"; then
  echo "Failed to connect to Sync Service. Using cached manifest..."
  use_cached_manifest
fi
```

```bash
# チェックサム検証とリトライ
download_with_checksum_verification() {
  local url="$1"
  local expected_checksum="$2"
  local output="$3"
  local max_retries=3

  for i in $(seq 1 $max_retries); do
    echo "Download attempt $i..."

    if curl -f -o "$output" "$url"; then
      local actual_checksum=$(sha256sum "$output" | cut -d' ' -f1)

      if [ "sha256:$actual_checksum" == "$expected_checksum" ]; then
        echo "Checksum verified successfully"
        return 0
      else
        echo "ERROR: Checksum mismatch (attempt $i)"
        echo "  Expected: $expected_checksum"
        echo "  Actual: sha256:$actual_checksum"
        rm -f "$output"
      fi
    else
      echo "ERROR: Download failed (attempt $i)"
    fi
  done

  echo "ERROR: All download attempts failed"
  return 1
}
```
